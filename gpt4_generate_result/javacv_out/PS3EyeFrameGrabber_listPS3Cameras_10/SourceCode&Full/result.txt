public static String[] listPS3Cameras() {
    int no = getCameraCount();
    String[] uuids;
    if (no > 0) {
        uuids = new String[no];
        for (--no; no >=0; no--) { uuids[no] = CLCamera.cameraUUID(no); }
        return uuids;
    }
    return null;
}
