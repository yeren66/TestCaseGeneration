public static double homogToRt(CvMat H,
        CvMat R1, CvMat t1, CvMat n1,

    CvMat S = S3x3.get(), U = U3x3.get(), V = V3x3.get();
    cvSVD(H, S, U, V, 0);
    double zeta = homogToRt(S, U, V, R1, t1, n1, R2, t2, n2);

}
