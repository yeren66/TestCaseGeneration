private Node rotateLeft(Node a) {
    Node b = a.right;
    b.parent = a.parent;

    a.right = b.left;

    if (a.right != null) {
        a.right.parent = a;
    }

    b.left = a;
    a.parent = b;

    if (b.parent != null) {
        if (b.parent.right == a) {
            b.parent.right = b;
        } else {
            b.parent.left = b;
        }
    }

    setBalance(a, b);

    return b;
}
