public HeapNode insert(int key) {
    HeapNode toInsert = new HeapNode(key); // creates the node
    if (this.empty()) {
        this.min = toInsert;
    } else { // tree is not empty
        min.setNext(toInsert);
        this.updateMin(toInsert);
    }
    this.numOfHeapNodes++;
    this.numOfTrees++;
    return toInsert;
}
